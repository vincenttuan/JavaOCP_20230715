【不重複的3位數】
問題描述：
0~9這10個數字可以組成多少個內容不重複的3位數。
例如:123為不重複,122為重複。

public class Contest04 {

	public static void main(String[] args) {
		
		int[] num = new int[3];
		int count = 1000;
		
		for(int i = 0; i <= 9; i++) {
			num[0] = i;
			for(int j = 0; j <= 9; j++) {
				num[1] = j;
				for(int k = 0; k <= 9; k++) {
					num[2] = k;
					if(num[0] == 0) {
						count--;
					}else if((num[0] == num[1]) || (num[0] == num[2]) || (num[1] == num[2])) {
						count--;
					}else {
						System.out.print(num[0] + "" + num[1] + num[2] + " ");
					}
				}
			}
		}
		
		System.out.println("一共有" + count + " 個這樣的數字");
	}

}
